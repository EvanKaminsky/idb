# GCP Flash + App Engine Reference
https://cloud.google.com/appengine/docs/standard/python/getting-started/python-standard-env

# Launch virtual environment
source env/bin/activate

# Exit virtual environment
deactivate

# Installing requirements using a Homebrewed Python
python2 -m pip install -t lib --install-option="--prefix=" -r requirements.txt

# Be sure you have the necessary GCP things installed:
# Follow instructtions here for Google Cloud SDK installation and initialization: 
https://cloud.google.com/sdk/docs/

# Get the App Engine extension for Python
gcloud components install app-engine-python

# Start the local dev server on Port 8080; use this to test
dev_appserver.py app.yaml

# Be sure you're a contributer for the project so that you're able to deploy

# Deploy app; this will take awhile and you'll have to sync the project the first time
gcloud app deploy

# View deployment
gcloud app browse
